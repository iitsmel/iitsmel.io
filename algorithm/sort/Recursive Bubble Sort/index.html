Recursive Bubble Sort

A quick review of bubble sort:
“Bubble Sort” swaps the first two elements if the first element is larger than the second one, the algorithm keeps doing those same steps, 2 and 3 swap, 3 and 4 swap, until all elements are in order.
Moving on to the word 'recursion'.
A recursive function is an independent function outside the main function. Which also called itself in its function.
example:
void recursion(){
recursion() ;
}

Method

When implementing Recursive Bubble Sort, one will need to use recursive function.

Usage & Code Demo

Recursive Bubble Sort is more like a way to test whether one is familiar with both recursion and bubble sort.

using std::cout;
using std::swap;

void recursivebubblesort(int after[], int number) {
 int temp = number - 1;
 for ( int i = 0 ; temp > i ; i++ ) {
  if (after[i] > after[i+1])
   swap(after[i], after[i+1]);
 }

 if (temp > 1 )
  recursivebubblesort(after, number-1);
}

void printout(int original[], int after[], int number) {
 cout << "Before Recursive Bubble Sort" << "\n";
 for (int i = 0 ; number > i ; i++ )
  cout << original[i] << "\n";
 cout << "After Recursive Bubble Sort" << "\n";
 for (int i = 0 ; number > i ; i++ )
  cout << after[i] << "\n";
}

Java example

import java.util.Arrays; public class RecursiveBubble {
 static void swapping(int after[], int i){
  int temp = after[i];
  after[i] = after[i+1];
  after[i+1] = temp;
 }

 static void recursivebubblesort(int after[], int length){
  for (int i = 0; length - 1 > i ; i++)
   if (after[i] > after[i+1])
    swapping( after, i );

  if ( length > 1 )
   recursivebubblesort(after, length - 1);
 }

 public static void main(String[] args){
  int original[] = { 5, 3, 2, 4, 1 };
  int after[] = { 5, 3, 2, 4, 1 };
  recursivebubblesort(after, after.length);
  System.out.println( "Original order" );
  System.out.println(Arrays.toString(original));
  System.out.println( "After Recursive Bubble Sort" );
  System.out.println(Arrays.toString(after));
 }
}

def swap(after, i):
 temp = after[i]
 after[i] = after[i+1]
 after[i+1] = temp

def recursivebubblesort(after, length):
 for i in range(length - 1):
  if after[i] > after[i + 1]:
   swap(after, i)
 if length > 1:
  recursivebubblesort(after, length-1)
